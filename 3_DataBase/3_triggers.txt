AFTER UPDATE ON usuario
FOR EACH ROW
BEGIN

id_usuario
)

'ACTUALIZACION USUARIO',
'Se modificó el estado del usuario',
NEW.id_usuario
);

END //

CREATE TRIGGER trg_validar_nota
BEFORE INSERT ON calificacion
FOR EACH ROW
BEGIN


IF NEW.nota < 0 OR NEW.nota > 20 THEN

SIGNAL SQLSTATE '45000'

SET MESSAGE_TEXT =
'La nota debe estar entre 0 y 20';

END IF;


END //

BEFORE INSERT ON pago_estudiante

'El monto debe ser mayor a cero';
